interface CalmCluster { lat: number; lng: number; count: number }

Helper:
  function computeCalmClusters(places: PlaceWithSensory[]): CalmCluster[] {
    const calmPlaces = places.filter(p => {
      if (!p.sensory) return false;
      const avg = (p.sensory.avg_sound + p.sensory.avg_light + p.sensory.avg_crowd) / 3;
      return avg <= 3;
    });
    // Simple greedy clustering: 0.008 degree proximity
    const visited = new Set<string>();
    const clusters: CalmCluster[] = [];
    for (const p of calmPlaces) {
      if (visited.has(p.place_id)) continue;
      const nearby = calmPlaces.filter(q => {
        if (visited.has(q.place_id)) return false;
        return Math.abs(q.geometry.location.lat - p.geometry.location.lat) < 0.008
          && Math.abs(q.geometry.location.lng - p.geometry.location.lng) < 0.008;
      });
      if (nearby.length >= 3) {
        nearby.forEach(q => visited.add(q.place_id));
        const lat = nearby.reduce((s, q) => s + q.geometry.location.lat, 0) / nearby.length;
        const lng = nearby.reduce((s, q) => s + q.geometry.location.lng, 0) / nearby.length;
        clusters.push({ lat, lng, count: nearby.length });
      }
    }
    return clusters;
  }

    const API_KEY = process.env.EXPO_PUBLIC_GOOGLE_MAPS_API_KEY;
    if (!API_KEY) return;
    setDetailsLoading(true);
    try {
      const url = `https://maps.googleapis.com/maps/api/place/details/json?place_id=${placeId}&fields=business_status,formatted_phone_number,opening_hours,current_opening_hours,rating,user_ratings_total,price_level&key=${API_KEY}`;

    } catch { /* silent */ }
    finally { setDetailsLoading(false); }
  }
